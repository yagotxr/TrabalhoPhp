CREATE TABLE User (
	id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	nome VARCHAR(30) NOT NULL,
	ultimoNome VARCHAR(30) NOT NULL,
	cpf VARCHAR(15),
	rg VARCHAR(15),
	dataNascimento VARCHAR(10),
	idade INT(3),
	endereco VARCHAR(250),
	user VARCHAR(20),
	pass VARCHAR(20),
	email VARCHAR(50),
	reg_date TIMESTAMP
)
